    /*
     * Tree Node
     */
  
    template<class T>
    class BinaryTree;
  
    template<class T>
    class TreeNode {
        friend BinaryTree<T>;
    public:
        TreeNode()
        :parent_(nullptr), leftChild_(nullptr), rightChild_(nullptr) {
            key_ = std::shared_ptr<T>(new T());
        }
        TreeNode(const T key)
        :parent_(nullptr), leftChild_(nullptr), rightChild_(nullptr) {
            key_ = std::shared_ptr<T>(new T(key));
        }
        ~TreeNode() {
            key_ = nullptr;
            parent_ = nullptr;
            leftChild_ = nullptr;
            rightChild_ = nullptr;
        }
  
        const T key() const {
            return *key_;
        }
        const std::shared_ptr<TreeNode>& parent() const {
            return parent_;
        }
        const std::shared_ptr<TreeNode>& leftChild() const {
            return leftChild_;
        }
        const std::shared_ptr<TreeNode>& rightChild() const {
            return rightChild_;
        }
  
    private:
        std::shared_ptr<T> key_;
        std::shared_ptr<TreeNode> parent_;
        std::shared_ptr<TreeNode> leftChild_, rightChild_;
    };
  
    /*
     * BinaryTree
     */
  
    template<class T>
    class BinaryTree {
    public:
        BinaryTree()
        :root_(nullptr) { }
        ~BinaryTree() { }
  
        void InitBinaryTree(const vector<shared_ptr<TreeNode<T>>>& vec) {
            for (int i = 0; i < vec.size(); i++) {
                auto treeNode = vec[i];
                if (vec[i]->key() != 0) {
                    if (((2 * i + 1) < vec.size()) && (vec[2 * i + 1]->key() != 0)) {
                        treeNode->leftChild_ = vec[2 * i + 1];
                    }
                    if (((2 * i + 2) < vec.size()) && (vec[2 * i + 2]->key() != 0)) {
                        treeNode->rightChild_ = vec[2 * i + 2];
                    }
                }
            }
            root_ = vec[0];
        }
  
        void NRPreorderTraverse() const {
            stack<std::shared_ptr<TreeNode<T>>> s;
            std::shared_ptr<TreeNode<T>> output;
            s.push(root_);
            while (!s.empty()) {
                output = s.top(); s.pop();
                cout << output->key() << " ";
                if (output->rightChild() != nullptr) {
                    s.push(output->rightChild());
                }
                if (output->leftChild() != nullptr) {
                    s.push(output->leftChild());
                }
            }
        }
  
        void NRInorderTraverse() const {
            stack<std::shared_ptr<TreeNode<T>>> s;
            std::shared_ptr<TreeNode<T>> output;
            auto p = root_;
            while ((p != nullptr) || (!s.empty())) {
                if (p != nullptr) {
                    s.push(p);
                    p = p->leftChild();
                }
                else {
                    output = s.top(); s.pop();
                    cout << output->key() << " ";
                    p = output->rightChild();
                }
            }
        }
  
        void NRPostorderTraverse() const {
            stack<std::shared_ptr<TreeNode<T>>> s;
            stack<std::shared_ptr<TreeNode<T>>> s_o;
            std::shared_ptr<TreeNode<T>> output;
            s.push(root_);
            while (!s.empty()) {
                output = s.top(); s.pop();
                s_o.push(output);
                if (output->leftChild() != nullptr) {
                    s.push(output->leftChild());
                }
                if (output->rightChild() != nullptr) {
                    s.push(output->rightChild());
                }
            }
            while (!s_o.empty()) {
                output = s_o.top(); s_o.pop();
                cout << output->key() << " ";
            }
        }
  
    private:
        shared_ptr<TreeNode<T>> root_;
    };
  
    int main(int argc, const char * argv[]) {
        BinaryTree<int> bsTree;
        vector<shared_ptr<TreeNode<int>>> vec;
        int total, key;
        cin >> total;
        for (int i = 0; i < total; i++) {
            cin >> key;
            auto treeNode = shared_ptr<TreeNode<int>>(new TreeNode<int>(key));
            vec.push_back(treeNode);
        }
        bsTree.InitBinaryTree(vec);
        bsTree.NRPreorderTraverse();
        cout << endl;
        bsTree.NRInorderTraverse();
        cout << endl;
        bsTree.NRPostorderTraverse();
        cout << endl;
  
        return 0;
    }
