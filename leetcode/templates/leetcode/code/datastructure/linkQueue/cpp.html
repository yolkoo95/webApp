  11  template<class T>
  12  class LinkQueue {
  13  public:
  14      LinkQueue()
  15          :front_(nullptr), rear_(nullptr), length_(0)
  16      {}
  17      virtual ~LinkQueue() {
  18          clear();
  19          front_ = nullptr;
  20          rear_ = nullptr;
  21          length_ = 0;
  22      }
  23  
  24      std::shared_ptr<Node<T>> front() {
  25          return front_;
  26      }
  27      const std::shared_ptr<Node<T>> front() const {
  28          return front_;
  29      }
  30      std::shared_ptr<Node<T>> rear() {
  31          return rear_;
  32      }
  33      const std::shared_ptr<Node<T>> rear() const {
  34          return rear_;
  35      }
  36      size_t length() {
  37          return length_;
  38      }
  39      size_t length() const {
  40          return length_;
  41      }
  42  
  43      void EnQueue(const T& element);
  44      const T& DeQueue();
  45      void Traverse(std::function<void (const T&)> visit);
  46      bool TestNull();
  47  
  48      // F type can be defined as: std::funciton<void (...)>, means function poer. Map will return a new queue which is copied from origin queue and influenced by function you designed.
  49      template <typename F>
  50      LinkQueue* Map(F visit) const{
  51          LinkQueue* new_queue = new LinkQueue();
  52          for (auto p = front_; p != nullptr; p = p->next()) {
  53              new_queue->EnQueue(visit(p->element()));
  54          }
  55          return new_queue;
  56      }
  57  
  58  protected:
  59      // Test Function;
  60      void Test();
  61  
  62  private:
  63      std::shared_ptr<Node<T>> front_;
  64      std::shared_ptr<Node<T>> rear_;
  65      size_t length_;
  66  
  67      bool isEmpty();
  68      bool isEmpty() const;
  69      void clear();
  70  };
  71  
  72  #endif

  89      // To use a test fixture, derive a class from testing::Test.
  90      class LinkQueueTest : public testing::Test {
  91      protected:
  92          virtual void SetUp() {
  93  
  94          }
  95  
  96          // Change age to double;
  97          static const Entry& DoubleAgeInEntry(Entry entry) {
  98              auto en = std::shared_ptr<Entry>(new Entry(entry));
  99              en->SetEntry(en->name(), en->age()*2);
  100              return *en;
  101          }
  102          // A helper function for test Queue::Map();
  103          // Don't forget new/delete;
  104          void MapTester(const LinkQueue<Entry>& q) {
  105              const LinkQueue<Entry> * const new_q = q.Map<function<const Entry&(Entry entry)>>(DoubleAgeInEntry);
  106              // Verifies that the new queue has the same size as q;
  107              ASSERT_EQ(q.length(), new_q->length());
  108  
  109              // Verifies whether Map() effects;
  110              for (auto p = q.front(), p_new = new_q->front(); p != nullptr; p = p->next(), p_new = p_new->next()) {
  111                  EXPECT_EQ(p->element().age()*2, p_new->element().age());
  112              }
  113  
  114              delete new_q;
  115          }
  116  
  117          LinkQueue<Entry> q0_;
  118          LinkQueue<Entry> q1_;
  119          LinkQueue<Entry> q2_;
  120      };
  121  
  122      // When you have a test fixture, you define a test using TEST_F
  123      // instead of TEST. Every TEST_F will restart SetUp()
  124      TEST_F(LinkQueueTest, Initializer) {
  125          EXPECT_EQ(0, q0_.length());
  126          EXPECT_TRUE(q0_.TestNull());
  127      }
  128  
  129      TEST_F(LinkQueueTest, DeQueue) {
  130          Entry en1("Donald Trump", 1);
  131          Entry en2("Quminzhi", 2);
  132          Entry en3("Taozhe", 3);
  133          q1_.EnQueue(en1);
  134          q2_.EnQueue(en2);
  135          q2_.EnQueue(en3);
  136  
  137          auto del_element = q1_.DeQueue();
  138          EXPECT_EQ(en1.name(), del_element.name());
  139          EXPECT_EQ(en1.age(), del_element.age());
  140          EXPECT_EQ(0, q1_.length());
  141  
  142          del_element = q2_.DeQueue();
  143          EXPECT_EQ(en2.name(), del_element.name());
  144          EXPECT_EQ(en2.age(), del_element.age());
  145          EXPECT_EQ(1, q2_.length());
  146      }
