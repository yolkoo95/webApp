  9      #define INT_MAX 2147483647
  10      
  11      int threeSumClosest(vector<int> &num, int target) {
  12          //sort the array
  13          sort(num.begin(), num.end());
  14      
  15          int n = num.size();
  16          int distance = INT_MAX;
  17          int result;
  18      
  19          for (int i=0; i<n-2; i++) {
  20              //skip the duplication
  21              if (i>0 && num[i-1]==num[i]) continue;
  22              int a = num[i];
  23              int low = i+1;
  24              int high = n-1;
  25              //convert the 3sum to 2sum problem
  26              while ( low < high ) {
  27                  int b = num[low];
  28                  int c = num[high];
  29                  int sum = a+b+c;
  30                  if (sum - target == 0) {
  31                      //got the final soultion
  32                      return target;
  33                  } else {
  34      
  35                      //tracking the minmal distance
  36                      if (abs(sum-target) < distance ) {
  37                          distance = abs(sum - target);
  38                          result = sum;
  39                      }
  40      
  41                      if (sum -target> 0) {
  42                          //skip the duplication
  43                          while(high>0 && num[high]==num[high-1]) high--;
  44                          //move the `high` pointer
  45                          high--;
  46                      } else {
  47                          //skip the duplication
  48                          while(low<n && num[low]==num[low+1]) low++;
  49                          //move the `low` pointer
  50                          low++;
  51                      }
  52                  }
  53              }
  54          }
  55      
  56          return result;
  57      }

  65          vector<int> n(a, a+sizeof(a)/sizeof(int));
  66          int target = 1;
  67          cout << threeSumClosest(n, target) << endl;
  68          return 0;
