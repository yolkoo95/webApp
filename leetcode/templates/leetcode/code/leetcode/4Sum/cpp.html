  8      vector<vector<int>> fourSum(vector<int> &num, int target) {
  9          vector< vector<int>> result;
  10          if (num.size()<4) return result;
  11          sort( num.begin(), num.end() );
  12      
  13          for(int i=0; i<num.size()-3; i++) {
  14              //skip the duplication
  15              if (i>0 && num[i-1]==num[i]) continue;
  16              vector<int> n(num.begin()+i+1, num.end());
  17              vector<vector<int>> ret = threeSum(n, target-num[i]);
  18              for(int j=0; j<ret.size(); j++){
  19                  ret[j].insert(ret[j].begin(), num[i]);
  20                  result.push_back(ret[j]);
  21              }
  22          }
  23      
  24          return result; 
  25      }

  27      vector<vector<int>> threeSum(vector<int> num, int target) {
  28      
  29          vector< vector<int>> result;
  30          //sort the array (if the qrray is sorted already, it won't waste any time)
  31          sort(num.begin(), num.end());
  32      
  33          int n = num.size();
  34      
  35          for (int i=0; i<n-2; i++) {
  36              //skip the duplication
  37              if (i>0 && num[i-1]==num[i]) continue;
  38              int a = num[i];
  39              int low = i+1;
  40              int high = n-1;
  41              while ( low < high ) {
  42                  int b = num[low];
  43                  int c = num[high];
  44                  if (a+b+c == target) {
  45                      //got the soultion
  46                      vector<int> v;
  47                      v.push_back(a);
  48                      v.push_back(b);
  49                      v.push_back(c);
  50                      result.push_back(v);
  51                      // Continue search for all triplet combinations summing to zero.
  52                      //skip the duplication
  53                      while(low<n && num[low]==num[low+1]) low++;
  54                      while(high>0 && num[high]==num[high-1]) high--;
  55                      low++;
  56                      high--;
  57                  } else if (a+b+c > target) {
  58                      //skip the duplication
  59                      while(high>0 && num[high]==num[high-1]) high--;
  60                      high--;
  61                  } else{
  62                      //skip the duplication
  63                      while(low<n && num[low]==num[low+1]) low++;
  64                      low++;
  65                  }
  66              }
  67          }
  68          return result;
  69      }
