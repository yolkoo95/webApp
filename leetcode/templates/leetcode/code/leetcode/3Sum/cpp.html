  8    vector<vector<int>>threeSum(vector<int> &num) {
  9    
  10        vector< vector<int> > result;
  11    
  12        //sort the array, this is the key
  13        sort(num.begin(), num.end());
  14    
  15        int n = num.size();
  16    
  17        for (int i=0; i<n-2; i++) {
  18            //skip the duplication
  19            if (i>0 && num[i-1]==num[i]) continue;
  20            int a = num[i];
  21            int low = i+1;
  22            int high = n-1;
  23            while ( low < high ) {
  24                int b = num[low];
  25                int c = num[high];
  26                if (a+b+c == 0) {
  27                    //got the soultion
  28                    vector<int> v;
  29                    v.push_back(a);
  30                    v.push_back(b);
  31                    v.push_back(c);
  32                    result.push_back(v);
  33                    // Continue search for all triplet combinations summing to zero.
  34                    //skip the duplication
  35                    while(low<n-1 && num[low]==num[low+1]) low++; 
  36                    while(high>0 && num[high]==num[high-1]) high--; 
  37                    low++;
  38                    high--;
  39                } else if (a+b+c > 0) {
  40                    //skip the duplication
  41                    while(high>0 && num[high]==num[high-1]) high--;
  42                    high--;
  43                } else{
  44                    //skip the duplication
  45                    while(low<n-1 && num[low]==num[low+1]) low++;
  46                    low++;
  47                } 
  48            }
  49        }
  50        return result;
  51    }

  81    vector<vector<int> > combination(vector<int> &v, int k) {
  82    
  83        vector<vector<int> > result;
  84        vector<int> d;
  85        int n = v.size();
  86        for (int i=0; i<n; i++){
  87            d.push_back( (i<k) ? 1 : 0 );
  88        }
  89    
  90        //1) from the left, find the [1,0] pattern, change it to [0,1]
  91        //2) move all of the 1 before the pattern to the most left side
  92        //3) check all of 1 move to the right
  93        while(1){
  94            vector<int> tmp;
  95            for(int x=0; x<n; x++){
  96                if (d[x]) tmp.push_back(v[x]);
  97            }
  98            sort(tmp.begin(), tmp.end());
  99            result.push_back(tmp);
  100            //step 1), find [1,0] pattern
  101            int i;
  102            bool found = false;
  103            int ones =0;
  104            for(i=0; i<n-1; i++){
  105    
  106                if (d[i]==1 && d[i+1]==0){
  107                    d[i]=0; d[i+1]=1;
  108                    found = true;
  109                    //step 2) move all of right 1 to the most left side
  110                    for (int j=0; j<i; j++){
  111                        d[j]=( ones > 0 ) ? 1 : 0;
  112                        ones--;
  113                    }
  114                    break;
  115                }
  116                if (d[i]==1) ones++;
  117            }
  118            if (!found){
  119                break;
  120            }
  121    
  122        }
  123        return result;
  124    }
