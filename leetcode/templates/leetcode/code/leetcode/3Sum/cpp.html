  7      /*

  10    
  11      template<class T>
  12      class BinaryTree;
  13    
  14      template<class T>
  15      class TreeNode {
  16          friend BinaryTree<T>;
  17      public:
  18          TreeNode()
  19          :parent_(nullptr), leftChild_(nullptr), rightChild_(nullptr) {
  20              key_ = std::shared_ptr<T>(new T());
  21          }
  22          TreeNode(const T key)
  23          :parent_(nullptr), leftChild_(nullptr), rightChild_(nullptr) {
  24              key_ = std::shared_ptr<T>(new T(key));
  25          }
  26          ~TreeNode() {
  27              key_ = nullptr;
  28              parent_ = nullptr;
  29              leftChild_ = nullptr;
  30              rightChild_ = nullptr;
  31          }
  32    
  33          const T key() const {
  34              return *key_;
  35          }
  36          const std::shared_ptr<TreeNode>& parent() const {
  37              return parent_;
  38          }
  39          const std::shared_ptr<TreeNode>& leftChild() const {
  40              return leftChild_;
  41          }
  42          const std::shared_ptr<TreeNode>& rightChild() const {
  43              return rightChild_;
  44          }
  45    
  46      private:
  47          std::shared_ptr<T> key_;
  48          std::shared_ptr<TreeNode> parent_;
  49          std::shared_ptr<TreeNode> leftChild_, rightChild_;
  50      };
  51    
  52      /*
  53       * BinaryTree
  54       */
  55    
  56      template<class T>
  57      class BinaryTree {
  58      public:
  59          BinaryTree()
  60          :root_(nullptr) { }
  61          ~BinaryTree() { }
  62    
  63          void InitBinaryTree(const vector<shared_ptr<TreeNode<T>>>& vec) {
  64              for (int i = 0; i < vec.size(); i++) {
  65                  auto treeNode = vec[i];
  66                  if (vec[i]->key() != 0) {
  67                      if (((2 * i + 1) < vec.size()) && (vec[2 * i + 1]->key() != 0)) {
  68                          treeNode->leftChild_ = vec[2 * i + 1];
  69                      }
  70                      if (((2 * i + 2) < vec.size()) && (vec[2 * i + 2]->key() != 0)) {
  71                          treeNode->rightChild_ = vec[2 * i + 2];
  72                      }
  73                  }
  74              }
  75              root_ = vec[0];
  76          }
  77    
  78          void NRPreorderTraverse() const {
  79              stack<std::shared_ptr<TreeNode<T>>> s;
  80              std::shared_ptr<TreeNode<T>> output;
  81              s.push(root_);
  82              while (!s.empty()) {
  83                  output = s.top(); s.pop();
  84                  cout << output->key() << " ";
  85                  if (output->rightChild() != nullptr) {
  86                      s.push(output->rightChild());
  87                  }
  88                  if (output->leftChild() != nullptr) {
  89                      s.push(output->leftChild());
  90                  }
  91              }
  92          }
  93    
  94          void NRInorderTraverse() const {
  95              stack<std::shared_ptr<TreeNode<T>>> s;
  96              std::shared_ptr<TreeNode<T>> output;
  97              auto p = root_;
  98              while ((p != nullptr) || (!s.empty())) {
  99                  if (p != nullptr) {
  100                      s.push(p);
  101                      p = p->leftChild();
  102                  }
  103                  else {
  104                      output = s.top(); s.pop();
  105                      cout << output->key() << " ";
  106                      p = output->rightChild();
  107                  }
  108              }
  109          }
  110    
  111          void NRPostorderTraverse() const {
  112              stack<std::shared_ptr<TreeNode<T>>> s;
  113              stack<std::shared_ptr<TreeNode<T>>> s_o;
  114              std::shared_ptr<TreeNode<T>> output;
  115              s.push(root_);
  116              while (!s.empty()) {
  117                  output = s.top(); s.pop();
  118                  s_o.push(output);
  119                  if (output->leftChild() != nullptr) {
  120                      s.push(output->leftChild());
  121                  }
  122                  if (output->rightChild() != nullptr) {
  123                      s.push(output->rightChild());
  124                  }
  125              }
  126              while (!s_o.empty()) {
  127                  output = s_o.top(); s_o.pop();
  128                  cout << output->key() << " ";
  129              }
  130          }
  131    
  132      private:
  133          shared_ptr<TreeNode<T>> root_;
  134      };
